//Print most repitative character name and count.

  String name = "deepak test automation";
        name = name.replace(" ", ""); // remove spaces

        int maxCount = 0;
        char maxChar = ' ';

        for (int i = 0; i < name.length(); i++) {
            int count = 1;

            for (int j = i + 1; j < name.length(); j++) {
                if (name.charAt(i) == name.charAt(j)) {
                    count++;
                }
            }
            if (count > maxCount) {
                maxCount = count;
                maxChar = name.charAt(i);
            }
        }
        System.out.println("Most repeating character: " + maxChar);
        System.out.println("Repeating count: " + maxCount);

//Print Missing Number 

        int arr[] = {12, 14, 18};
        int missingNu = 0;

        int step = arr[1] - arr[0]; // dynamic step

        for (int i = 0; i < arr.length - 1; i++) {
            int diff = arr[i + 1] - arr[i];

            if (diff != step) {
                missingNu = arr[i] + step;
                break;
            }
        }

        System.out.println(missingNu);
